## -문법-
선택자 = selector
속성 = property ; 
값 = value
선언 = declaration
선언부 = {}
규칙 = rule set
주석 = /* ~ */

## -적용방식-
inline <div style='...'> 내용 </div> (자주사용x)
internal <style> ... </style>
external <link rel='stylesheet' href='css/style.css'> (주로 사용함)

## -class 선택자-
_css = .foo {font-size: 30px;}
_html = <p class='foo'> ... </p>

## -속성 선택자-
_css = p[class] {color: silver;}
_html = <p class='foo'>hello</p>
